Why Rate Matching?

In 5G NR, the channel encoder produces a fixed number of coded bits, but the physical layer resources available for transmission vary based on:

  • üì° Bandwidth allocation - Number of PRBs (Physical Resource Blocks)
  • ‚è±Ô∏è Time resources - Number of OFDM symbols
  • üìä Modulation scheme - QPSK, 16QAM, 64QAM, 256QAM
  • üîß MIMO layers - Spatial multiplexing
üí° Core Purpose: Rate matching adapts the encoder output to fit exactly into the allocated physical resources by repeating (if too few bits) or puncturing (if too many bits) the coded bits.

üßÆ Section 2: Mathematical & Algorithmic Model

Rate Matching Components

1. Bit Collection

Input: Coded bits from channel encoder
D = [d‚ÇÄ, d‚ÇÅ, d‚ÇÇ, ..., dN-1]
where N = encoder output length

2. Sub-block Interleaving

Rearranges bits to improve error resilience:

For LDPC (Base Graph 1):
P = [0, 2, 4, 6, ..., 1, 3, 5, 7, ...]

vi = dP[i]
Why interleave? Distributes consecutive bits across the transmission, making the system more robust against burst errors.

3. Circular Buffer

Core concept: Create a circular buffer of size Ncb where bits can be read from any starting position:

Circular Buffer: CB[0], CB[1], ..., CB[Ncb-1]

Read pointer: k0 (starting position)
Wraps around: CB[(k0 + i) mod Ncb]
Circular Buffer Construction:
1. Place interleaved bits into buffer
2. For LDPC: Ncb = N (no repetition within buffer)
3. For Polar: May include repetition for mother code

4. Bit Selection Logic

Given: E = target output length
N = encoder output length

If E > N: Repetition (some bits repeated)
If E < N: Puncturing (some bits omitted)
If E = N: Perfect match

5. Redundancy Version (RV)

For HARQ retransmissions, different starting positions provide incremental redundancy:

RV0: k‚ÇÄ = 0 (start from beginning)
RV1: k‚ÇÄ = ‚åà17√óNcb/66‚åâ
RV2: k‚ÇÄ = ‚åà33√óNcb/66‚åâ
RV3: k‚ÇÄ = ‚åà56√óNcb/66‚åâ

6. Effective Code Rate

Reff = K / E

where K = information bits
E = rate-matched output bits
‚ö†Ô∏è Important: Lower code rate (more redundancy) = better error protection but lower spectral efficiency. Rate matching allows dynamic trade-off based on channel conditions.

Rate Matching Algorithms

LDPC Rate Matching Implementation 
// LDPC Rate Matching for 5G NR

struct LDPCRateMatchParams {
    int baseGraph;        // BG1 or BG2
    int Z;                // Lifting size
    int N;                // Encoder output length
    int E;                // Target output length
    int rv;               // Redundancy version (0,1,2,3)
    int Qm;               // Modulation order
};

// Sub-block interleaving for LDPC
void ldpcSubblockInterleaving(uint8_t* input, uint8_t* output, int N) {
    // Interleaving pattern: even indices first, then odd
    int idx = 0;
    
    // Write even-indexed bits
    for (int i = 0; i < N; i += 2) {
        output[idx++] = input[i];
    }
    
    // Write odd-indexed bits
    for (int i = 1; i < N; i += 2) {
        output[idx++] = input[i];
    }
}

// Calculate circular buffer size
int calculateNcb(int N, int baseGraph) {
    if (baseGraph == 1) {
        return N;  // For BG1
    } else {
        return N;  // For BG2
    }
}

// Calculate starting position based on RV
int calculateRvStartPosition(int Ncb, int rv, int Qm) {
    int k0 = 0;
    
    switch (rv) {
        case 0:
            k0 = 0;
            break;
        case 1:
            k0 = (int)(17.0 * Ncb / 66.0);
            break;
        case 2:
            k0 = (int)(33.0 * Ncb / 66.0);
            break;
        case 3:
            k0 = (int)(56.0 * Ncb / 66.0);
            break;
    }
    
    // Align to modulation order
    k0 = (k0 / Qm) * Qm;
    return k0;
}

// Main LDPC rate matching function
void ldpcRateMatching(
    uint8_t* input,           // Coded bits from encoder
    uint8_t* output,          // Rate-matched output
    LDPCRateMatchParams params
) {
    // 1. Sub-block interleaving
    uint8_t* interleaved = new uint8_t[params.N];
    ldpcSubblockInterleaving(input, interleaved, params.N);
    
    // 2. Create circular buffer
    int Ncb = calculateNcb(params.N, params.baseGraph);
    
    // 3. Calculate starting position
    int k0 = calculateRvStartPosition(Ncb, params.rv, params.Qm);
    
    // 4. Bit selection from circular buffer
    for (int i = 0; i < params.E; i++) {
        int bufferIdx = (k0 + i) % Ncb;
        output[i] = interleaved[bufferIdx];
    }
    
    delete[] interleaved;
}

// Example usage
void exampleLDPC() {
    LDPCRateMatchParams params;
    params.baseGraph = 1;
    params.Z = 384;
    params.N = 17280;      // Encoder output
    params.E = 12000;      // Physical resources
    params.rv = 0;         // First transmission
    params.Qm = 4;         // 16QAM
    
    uint8_t* input = new uint8_t[params.N];
    uint8_t* output = new uint8_t[params.E];
    
    ldpcRateMatching(input, output, params);
    
    delete[] input;
    delete[] output;
}
Polar Rate Matching Implementation 
// Polar Rate Matching for 5G NR

struct PolarRateMatchParams {
    int K;                // Information bits
    int E;                // Target output length
    int nMax;             // Maximum code length (10)
    int iIL;              // Interleaving flag
    bool isPCCA;          // PCCA or PCCC
};

// Sub-block interleaving for Polar (optional)
void polarSubblockInterleaving(uint8_t* input, uint8_t* output, int N) {
    // Bit-reversal permutation pattern
    int n = 0;
    while ((1 << n) < N) n++;
    
    for (int i = 0; i < N; i++) {
        int rev = 0;
        for (int j = 0; j < n; j++) {
            if (i & (1 << j)) {
                rev |= (1 << (n - 1 - j));
            }
        }
        if (rev < N) {
            output[rev] = input[i];
        }
    }
}

// Calculate mother code length
int calculateMotherCodeLength(int K, int E, int nMax) {
    int n = 0;
    
    // Find minimum n such that 2^n >= E
    while ((1 << n) < E) {
        n++;
    }
    
    // Ensure n <= nMax
    if (n > nMax) n = nMax;
    
    return (1 << n);
}

// Rate matching with repetition/puncturing
void polarRateMatching(
    uint8_t* input,           // Polar encoded bits
    uint8_t* output,          // Rate-matched output
    PolarRateMatchParams params
) {
    // 1. Calculate mother code length N
    int N = calculateMotherCodeLength(params.K, params.E, params.nMax);
    
    // 2. Sub-block interleaving (if enabled)
    uint8_t* interleaved = new uint8_t[N];
    if (params.iIL == 1) {
        polarSubblockInterleaving(input, interleaved, N);
    } else {
        memcpy(interleaved, input, N);
    }
    
    // 3. Determine rate matching type
    if (params.E >= N) {
        // Repetition case
        for (int k = 0; k < params.E; k++) {
            output[k] = interleaved[k % N];
        }
    } else {
        // Puncturing/shortening case
        if (params.E <= (3 * N / 4)) {
            // Shortening
            int numShorten = N - params.E;
            for (int k = 0; k < params.E; k++) {
                output[k] = interleaved[k + numShorten];
            }
        } else {
            // Puncturing
            int numPuncture = N - params.E;
            int outIdx = 0;
            for (int k = 0; k < N; k++) {
                if (k >= numPuncture) {
                    output[outIdx++] = interleaved[k];
                }
            }
        }
    }
    
    delete[] interleaved;
}

// Example usage
void examplePolar() {
    PolarRateMatchParams params;
    params.K = 128;        // Information bits
    params.E = 256;        // Physical resources
    params.nMax = 10;      // Max mother code 2^10 = 1024
    params.iIL = 1;        // Enable interleaving
    params.isPCCA = true;  // PCCA type
    
    int N = calculateMotherCodeLength(params.K, params.E, params.nMax);
    uint8_t* input = new uint8_t[N];
    uint8_t* output = new uint8_t[params.E];
    
    polarRateMatching(input, output, params);
    
    delete[] input;
    delete[] output;
}
Common Utility Functions 
// Common rate matching utilities

// Calculate effective code rate
double calculateCodeRate(int K, int E) {
    return (double)K / (double)E;
}

// Calculate spectral efficiency
double calculateSpectralEfficiency(int K, int E, int Qm) {
    double codeRate = calculateCodeRate(K, E);
    return codeRate * Qm;  // bits/symbol
}

// Determine operation type
enum RateMatchType {
    REPETITION,
    PUNCTURING,
    SHORTENING,
    PERFECT_MATCH
};

RateMatchType determineRateMatchType(int N, int E) {
    if (E > N) {
        return REPETITION;
    } else if (E < N) {
        // Additional logic for puncturing vs shortening
        if (E <= (3 * N / 4)) {
            return SHORTENING;
        } else {
            return PUNCTURING;
        }
    } else {
        return PERFECT_MATCH;
    }
}

// Print rate matching statistics
void printRateMatchStats(int K, int N, int E, int Qm) {
    printf("=== Rate Matching Statistics ===\n");
    printf("Information bits (K): %d\n", K);
    printf("Encoder output (N): %d\n", N);
    printf("Target output (E): %d\n", E);
    printf("Modulation order (Qm): %d\n", Qm);
    
    double codeRate = calculateCodeRate(K, E);
    printf("Effective code rate: %.3f\n", codeRate);
    
    double spectralEff = calculateSpectralEfficiency(K, E, Qm);
    printf("Spectral efficiency: %.3f bits/symbol\n", spectralEff);
    
    RateMatchType type = determineRateMatchType(N, E);
    printf("Operation: ");
    switch (type) {
        case REPETITION:
            printf("REPETITION (%d bits repeated)\n", E - N);
            break;
        case PUNCTURING:
            printf("PUNCTURING (%d bits punctured)\n", N - E);
            break;
        case SHORTENING:
            printf("SHORTENING (%d bits shortened)\n", N - E);
            break;
        case PERFECT_MATCH:
            printf("PERFECT MATCH\n");
            break;
    }
    printf("==============================\n");
}
